// Global floating handler included in library document.addEventListener('click', (e) => { const trigger = e.target.closest('[zeno-floating]'); if (trigger) { const id = trigger.getAttribute('zeno-floating'); const widget = document.querySelector(id); if (widget) { widget.style.display = 'block'; // Position widget near trigger... } } });
